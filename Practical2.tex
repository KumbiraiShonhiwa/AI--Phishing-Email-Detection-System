\subsection{Network Scanning and Reconnaissance (5 Marks)}
\begin{enumerate}
    \item 	extbf{Discover the target's IP address:}
    \begin{lstlisting}[language=bash]
nmap -sn 192.168.1.0/24
    \end{lstlisting}
    \item 	extbf{Perform a full port scan:}
    \begin{lstlisting}[language=bash]
nmap -p- -sV <TARGET-IP>
    \end{lstlisting}
    \item 	extbf{Scan the web application for vulnerabilities:}
    \begin{lstlisting}[language=bash]
nikto -h http://<TARGET-IP>
    \end{lstlisting}
\end{enumerate}
